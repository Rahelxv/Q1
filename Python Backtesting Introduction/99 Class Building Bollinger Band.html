import pandas as pd
import numpy as np
import yfinance as yf
from itertools import product

class MRBollingerBands:

  #create innitiaion function
  def __init__(self, symbol, start, end, interval, ptc, sma, std):
    self.symbol = symbol
    self.start = start
    self.end = end
    #interval or frequency of data
    self.interval = interval
    #proportion trading cost
    self.ptc = ptc
    #take the initial ma
    self.sma = sma
    #take the initial std
    self.std = std
    #get data run
    self.get_data()
    #storing all the variable that been made
    self.results = None

  #creating get data function
  def get_data(self):
    #downloading data from yahoo finance
    data = yf.download(self.symbol, self.start, self.end, interval=self.interval)
    #taking only price values
    data = data['Close']
    #rename close to price
    data = data.rename(columns={self.symbol: 'price'})
    #create return for given data
    data['returns'] = np.log(data.div(data.shift(1)))
    #droping missing values
    data.dropna(inplace = True)
    #save the data
    self.data = data

  #create backtesting function
  def test_strategy(self):
    #getting data from get data function
    data = self.data.copy().dropna()
    #set variable for given variable
    SMA = self.sma
    std = self.std
    #create SMA that we choose
    data['SMA'] = data['price'].rolling(SMA).mean()
    #lower band of bollinger
    data['Lower'] = data['SMA'] - data['price'].rolling(SMA).std() * std
    #upper band of bollinger
    data['Upper'] = data['SMA'] + data['price'].rolling(SMA).std() * std
    #calculated distance of sma and price to know the position of price relative to the sma
    data['distance'] = data.price - data.SMA
    #create position for oversold -> we go long
    data['position'] = np.where(data.price < data.Lower, 1, np.nan)
    #create position for overbought -> we go short
    data['position'] = np.where(data.price > data.Upper, -1, data['position'])
    #natural position (if crossing middle band or sma close the position, this telling us if the position change)
    data['position'] = np.where(data.distance * data.distance.shift(1) < 0, 0, data['position'])
    #filling forward the still NaN values and set first NaN as 0
    data.position = data.position.ffill().fillna(0)
    #calculate backtest result
    data['strategy'] = data.position.shift(1) * data['returns']
    #droping missing values
    data.dropna(inplace =True)
    #calculated banckmark returns hold and buy & strategy cumulative returns
    data['creturns'] = data['returns'].cumsum().apply(np.exp)
    data['cstrategy'] = data['strategy'].cumsum().apply(np.exp)
    #adding trading cost
    ptc = self.ptc
    #calculated the number of trades
    data['trades'] = data.position.diff().fillna(0).abs()
    #strategy net profit after trading cost
    data['strategy_net'] = data.strategy - data.trades * ptc
    #calculated cumulative returns (if we invested $1)
    data['cstrategy_net'] = data.strategy_net.cumsum().apply(np.exp)
    #storing the values on self.results
    self.results = data

    #calculatinng strategy evaluation or performance
    perf = data['cstrategy_net'].iloc[-1]#absolute performance
    #outperformance (cumulative performance of strategy - cumulative benchmark strategy) which result of how poor or good the strategy compere to the banchmark
    outperf = perf - data['creturns'].iloc[-1]
    #return of round 6 digit of performance (1) & outperfomance of strategy compare to banchmark (2)
    return round(perf, 6), round(outperf, 6)

  #function to plot the result
  def plot_result(self):
    #if not run backtest, make sure to remember run backtest first
    if self.results is None:
      print('Run test_strategy() first.')
    else:
      #plotting the benchmark and the strategy return
      title = '{} | Moving Average of {} & Standard Deviation of {}'.format(self.symbol, self.sma, self.std)
      self.results[['creturns', 'cstrategy_net']].plot(title = title, figsize = (12,8))

  #optimization function
  def optimization_parameter(self, ma_range, std_range):
    #unpack the range combination with (*) and store all the combination in combinations variable
    combinations = list(product( range(*ma_range), range(*std_range)))

    #iterate for backtesting and store all the backtesting result in list
    results = []
    for comb in combinations:
      self.sma = comb[0]
      self.std = comb[1]
      results.append(self.test_strategy()[0])

    #saving the best performance in variable and taking the argument or combination which resulting in best performance
    best_perf = np.max(results) #best performance
    opt = combinations[np.argmax(results)] #optimal combination parameters

    #create a df with many results
    many_results = pd.DataFrame(data = combinations, columns = ['SMA', 'STD']) #storing all the combinations in many_results variable
    many_results['performance'] = results #storing the performance in performance column
    self.results_overview = many_results #storing the data frame into result overview function (could show it if you want to see the list

    return best_perf, opt
