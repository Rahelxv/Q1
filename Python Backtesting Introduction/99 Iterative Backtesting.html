#this is agnostic class which we could use for any strategy

#define class
class IterativeBase():

    #define innit function
    def __init__(self, symbol, start, end, amount, use_spread = True):
        self.symbol = symbol
        self.start = start
        self.end = end
        self.initial_balance = amount
        self.current_balance = amount
        self.units = 0
        self.trades = 0
        self.position = 0
        self.use_spread = use_spread
        self.get_data()

    #define get data function
    def get_data(self):
        raw = pd.read_csv('detailed.csv', parse_dates=['time'], index_col='time').dropna()
        raw = raw.loc[self.start:self.end]

        #create returns and safe all the data into self.data
        raw['returns'] = np.log(raw.price / raw.price.shift(1))
        self.data = raw

    #plotting the data function
    def plot_data(self, cols = None):
        
        #if the column price is None, make it 'price' and what ever column we specify we plot
        if cols is None:
            cols = 'price'
        self.data[cols].plot(figsize = (12, 8), title = self.symbol)

    #get the value function (data, price, spread) for given bar we specify
    def get_values(self, bar):
        date = str(self.data.index[bar].date())
        price = round(self.data.price.iloc[bar], 5)
        spread = round(self.data.spread.iloc[bar], 5)
        return date, price, spread

    #Print current balance function
    def print_current_balance(self, bar):
        date, price, spread = self.get_values(bar)
        print('{} | Current Balance: {}'.format(date, round(self.current_balance, 2)))

    #function to buy an instrument; we could specify units or amount
    def buy_instrument(self, bar, units = None, amount = None):
        date, price, spread = self.get_values(bar)

        #trade cost 
        if self.use_spread:
            price += spread/2 #ask price

        #if amount specify
        if amount is not None: #use units if units are passed, otherwise calculate units
            units = int(amount / price)

        #if not specify
        self.current_balance -= units * price #reduce cash balance by 'purchase price'
        self.units += units
        self.trades += 1
        print('{} | Buying {} for {}'.format(date, units, round(price, 2)))

    #function to sell an instrument; we could specify units or amount
    def sell_instrument(self, bar, units = None, amount = None):
        date, price, spread = self.get_values(bar)

        #trade cost 
        if self.use_spread:
            price -= spread/2 #bid price

        #if amount specify
        if amount is not None: #use units if units are passed, otherwise calculate units
            units = int(amount / price)

        #if not specify
        self.current_balance += units * price #increase cash balance by 'purchase price'
        self.units -= units
        self.trades += 1
        print('{} | Selling {} for {}'.format(date, units, round(price, 2)))

    #print current position values
    def print_current_position_value(self, bar):
        date, price, spread = self.get_values(bar)
        cpv = self.units * price
        print('{} | Current Position Value = {}'.format(date, round(cpv, 2)))

    #print net values for given bar (floating p&l + account balance) function
    def print_current_nav(self, bar):
        date, price, spread = self.get_values(bar)
        nav = self.current_balance + self.units * price
        print('{} | Net Asset Value = {}'.format(date, round(nav, 2)))

    #close position function
    def close_pos(self, bar):
        date, price, spread = self.get_values(bar)
        
        #line for sperator
        print(75 * '-')
        
        #header
        print('{} | +++ CLOSING FINAL POSITION +++'.format(date))

        #calculte current balance when closing 
        self.current_balance += self.units * price #closing final position (works with short and long)

        #mid price close position and caculated the half spread (if we false to use spread then it 0 or deduct no cost, and if true then we deducted the cost)
        self.current_balance -=(abs(self.units)*spread/2*self.use_spread)

        #show date, units when closing happen, and what price are
        print('{} | closing position of {} for {}'.format(date, self.units, price))

        #set the unit to neutral and trade +1 which tell than trade activity happen when closing
        self.units = 0 #setting position to neutral
        self.trades += 1 

        #take performance percentace, current balance, and number of trade being made 
        perf = (self.current_balance - self.initial_balance) / self.initial_balance * 100 #this is the performance percentage calculation
        self.print_current_balance(bar)
        print(' {} | net performance (%) = {}'.format(date, round(perf, 2) ))
        print('{} | number of trades executed = {}'.format(date, self.trades))
        
        #line for sperator
        print(75 * '-')

#class for the strategy we want

#define class child
class IterativeBacktest(IterativeBase):

    #helper method
    def go_long(self, bar, units = None, amount = None):

        #if we use units
        if self.position == -1: #if our position in short now 
            self.buy_instrument(bar, units = -self.units)
        if units:
            self.buy_instrument(bar, units = units)

        #if we use amount
        elif amount:
            if amount == 'all': #if amount == 'all' when we specify, then we use all balance, other wise use amount we input
                amount = self.current_balance
            self.buy_instrument(bar, amount = amount)#if not we do given amount

    
    #helper method
    def go_short(self, bar, units = None, amount = None):
        #if we use units
        if self.position == 1: #if our position in long now 
            self.sell_instrument(bar, units = self.units)
        if units:
            self.sell_instrument(bar, units = units)

        #if we use amount
        elif amount:
            if amount == 'all': #if amount == 'all' when we specify, then we use all balance, other wise use amount we input
                amount = self.current_balance
            self.sell_instrument(bar, amount = amount)#if not we do given amount

    
    #strategy function
    def test_sma_strategy(self, SMA_S, SMA_L):

        #nice printout
        stm = 'Testing SMA strategy | {} | SMA_S = {} & SMA_L = {}'.format(self.symbol, SMA_S, SMA_L)
        print('-' * 75)
        print(stm)
        print('-' * 75)

        #reset the position and trades into 0 (neutral)
        self.position = 0 #initial neutral position 
        self.trades = 0 # no trades yet
        self.current_balance = self.initial_balance #reset initial capital
        self.get_data() #reset dataset

        #prepare data
        self.data['SMA_S'] = self.data['price'].rolling(SMA_S).mean()
        self.data['SMA_L'] = self.data['price'].rolling(SMA_L).mean()
        self.data.dropna(inplace = True)

        #sma crossover strategy
        for bar in range(len(self.data)-1): #all bar except the last bar
            if self.data['SMA_S'].iloc[bar] > self.data['SMA_L'].iloc[bar]: #signal to go long
                if self.position in [0, -1]: #if now poition == 0 or in short position
                    self.go_long(bar, amount = 'all') #go long with full amount # --example of 1% of capital (self.current_balance * 0.01)
                    self.position = 1 #long position
            elif self.data['SMA_S'].iloc[bar] < self.data['SMA_L'].iloc[bar]: #signal to go short
                if self.position in [0, 1]: #if now poition == 0 or in long position
                    self.go_short(bar, amount='all') #go short with full amount
                    self.position = -1 #short position
        self.close_pos(bar+1) #close position at the last bar
