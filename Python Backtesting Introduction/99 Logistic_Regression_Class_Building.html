#importin necessary library
import pandas as pd
import numpy as np
import yfinance as yf
from itertools import product
#import the mechine learning
from sklearn.linear_model import LogisticRegression
from sklearn.multiclass import OneVsRestClassifier
#importing accuracy tools
from sklearn.metrics import accuracy_score

#creat class
class MLBacktester:

  #create innitation
  def __init__(self, symbol, start, end, interval, tc):
    self.symbol = symbol
    self.start = start
    self.end = end
    self.tc = tc
    self.interval = interval
    self.results = None
    self.get_data()

  #create get data function
  def get_data(self):
    #downloading data from yahoo finance
    data = yf.download(self.symbol, self.start, self.end, interval=self.interval)
    #taking only price values
    data = data['Close']
    #rename close to price
    data = data.rename(columns={self.symbol: 'price'})
    #create return for given data
    data['returns'] = np.log(data.div(data.shift(1)))
    #droping missing values
    data.dropna(inplace = True)
    #save the data
    self.data = data

  #create backtesting function
  def test_strategy(self, lags):
    #accounted for trading cost
    ptc = self.tc
    #getting data from get data function
    data = self.data.copy().dropna()
    #create data direction return base on returns
    data['direction'] = np.sign(data.returns)
    #create list to store
    self.lags = lags
    #create list for store the name of lag
    cols = []
    #iterate to create lag varible and store the name in a list
    #range doesnt including a stop which 5 so we add +1
    for lag in range(1, lags + 1):
      col = 'lag{}'.format(lag)
      data[col] = data.returns.shift(lag)
      cols.append(col)
    #droping missing values of lag etc
    data.dropna(inplace = True)
    # take all data lag mean
    means = data[cols].mean()
    #takes all the lags standard deviation
    stand_devs = data[cols].std()
    #devided data lags - mean with standar deviation to creat standarized values of lags
    data[cols] = (data[cols] - means) / stand_devs
    #store the logistic regression into variable to be fitted
    lm = OneVsRestClassifier(LogisticRegression(C = 1e6, max_iter = 100000))
    #fittet the lag valuesto predict direction
    lm.fit(data[cols], data.direction)
    #create pred column to show the prediction vs actual retruns or direction
    data['pred'] = lm.predict(data[cols])
    #calculated score automaticly which we input the correct return direction vs prediction of direction with logistic regression
    accuracy = accuracy_score(y_true = data.direction, y_pred = data.pred)
    #make so we could see the percentace of correct prediction
    self.accuracy = accuracy
    #create strategy returns
    data['strategy'] = data.pred * data.returns
    #create cumulative return banckmark and strategy return
    data['creturns'] = data.returns.cumsum().apply(np.exp)
    data['cstrategy'] = data.strategy.cumsum().apply(np.exp)
    #create how many trades activity we takes
    data['trades'] = data.pred.diff().fillna(0).abs()
    #store trade taken so we could see how many trade were taken
    self.trade_takes = data.trades.value_counts()
    #strategy net profit after trading cost
    data['strategy_net'] = data.strategy - data.trades * ptc
    #calculated cumulative returns (if we invested $1)
    data['cstrategy_net'] = data.strategy_net.cumsum().apply(np.exp)
    #storing the values on self.results
    self.results = data

    #calculatinng strategy evaluation or performance
    perf = data['cstrategy_net'].iloc[-1]#absolute performance
    #outperformance (cumulative performance of strategy - cumulative benchmark strategy) which result of how poor or good the strategy compere to the banchmark
    #outperf = perf - data['creturns'].iloc[-1]
    #return of round 6 digit of performance (1) & outperfomance of strategy compare to banchmark (2)
    return round(perf, 6)

  #function to plot the result
  def plot_result(self):
    #if not run backtest, make sure to remember run backtest first
    if self.results is None:
      print('Run test_strategy() first.')
    else:
      #plotting the benchmark and the strategy return
      title = '{} | lags of {}'.format(self.symbol, self.lags)
      self.results[['creturns', 'cstrategy_net']].plot(title = title, figsize = (12,8))
