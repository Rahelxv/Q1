import numpy as np
import pandas as pd

#creating class for contrarian backtesting strategy
class ContrarianBacktester():
    
    #initiation input or variable needed
    def __init__(self, symbol, start, end, tc):
        self.symbol = symbol
        self.start = start
        self.end = end
        self.tc = tc
        self.results = None
        self.get_data()
        
    #getting the data
    def get_data(self):
        #get the data
        data = pd.read_csv('intraday_pairs.csv', parse_dates=['time'], index_col='time')
        #create symbol that we choose as data frame
        data = data[self.symbol].to_frame()
        #change the column name to price
        data = data.rename(columns={self.symbol : 'price'})
        #create the returns column for return windows
        data['returns'] = np.log(data.div(data.shift(1)))
        #droping NaN values
        data.dropna(inplace = True)
        #save data safe
        self.data = data

    #function for backtesting (we specify 1 windows for default if not specify)
    def test_strategy(self, window = 1):
        #saving window input in self.window for plotting
        self.window = window
        #getting data from get data function
        data = self.data.copy().dropna()
        #create position trade for given window we specify, for contrarian strategy
        data['position'] = -np.sign(data['returns'].rolling(window).mean()) #contrarian (minus sign)
        #calculated strategy return by multiply position and return mean of windows given
        data['strategy'] = data.position.shift(1) * data['returns']
        #droping NaN values
        data.dropna(inplace = True)

        #adding trading cost
        data['trades'] = data.position.diff().fillna(0).abs()
        data['strategy_net'] = data.strategy - data.trades * self.tc
        
        
        #calulated result of banckmark and strategy with trading cost
        data['creturns'] = data['returns'].cumsum().apply(np.exp)
        data['cstrategy_net'] = data.strategy_net.cumsum().apply(np.exp)

        
        #this is perfromance result without trading cost    #data['cstrategy'] = data['strategy'].cumsum().apply(np.exp)
        #store the backtest result into result 
        self.results = data

        #calculatinng strategy evaluation or performance
        perf = data['cstrategy_net'].iloc[-1]#absolute performance
        #outperformance (cumulative performance of strategy - cumulative benchmark strategy) which result of how poor or good the strategy compere to the banchmark
        outperf = perf - data['creturns'].iloc[-1] 
        #return of round 6 digit of performance (1) & outperfomance of strategy compare to banchmark (2)
        return round(perf, 6), round(outperf, 6)

    #function to plot the result
    def plot_result(self):
        if self.results is None:
            print('Run test_strategy() first.')
        else:
            #plotting the benchmark and the strategy return
            title = '{} | Windows of {}'.format(self.symbol, self.window)
            self.results[['creturns', 'cstrategy_net']].plot(title = title, figsize = (12,8))

    #create optimization function
    def optimize_parameter(self, start, end, step):
        # List to store performance results
        results = []
        #range to store the range for optimization
        windows = range(start, end, step)
        #create iteration loop for backtesting the given window
        for w in windows:
            #taking data given in get_data() to calculated the position and strategy given it need data['returns']
            data = self.data.copy()
            #create position and strategy return for every windows given
            data['position'] = -np.sign(data['returns'].rolling(w).mean())
            data['strategy'] = data['position'].shift(1) * data['returns']
            data.dropna(inplace=True)

            #adding trading cost
            data['trades'] = data.position.diff().fillna(0).abs()
            data['strategy_net'] = data.strategy - data.trades * self.tc
            
            #calculated performance sum of given window
            data['cstrategy_net'] = data.strategy_net.cumsum().apply(np.exp)
            perf = data['cstrategy_net'].iloc[-1]
    
            # Save performance result
            results.append({'window': w, 'performance': perf})
    
        #adding buyhold as banckmark to the 
        buyhold_perf = data['returns'].cumsum().apply(np.exp).iloc[-1]
        results.append({'window': 'buyhold', 'performance': round(buyhold_perf, 6)})
        
        # Convert to DataFrame
        performance_df = pd.DataFrame(results)

        #result overview to look at all the performance window
        self.results_overview = performance_df

        #show the best optimized windows and performance result
        self.wind_arg = performance_df.loc[performance_df['performance'].idxmax()]
        return  self.wind_arg
