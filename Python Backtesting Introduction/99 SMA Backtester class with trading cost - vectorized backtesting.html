import tpqoa
#import needed library to handel data
import numpy as np
import pandas as pd
from itertools import product

#creating class for SMA cross-over backtesting strategy
class SMABacktester():
    ''' Class for the vectorized backtesting of SMA-based trading strategies.
    '''
    #creating initiation of variable to start the class
    def __init__(self, symbol, SMA_S, SMA_L, start, end, freq, ptc):
        '''
        Parameters
        ----------
        symbol: str
            ticker symbol (instrument) to be backtested
        SMA_S: int
            moving window in bars (e.g. days) for shorter SMA
        SMA_L: int
            moving window in bars (e.g. days) for longer SMA
        start: str
            start date for data import
        end: str
            end date for data import
        freq: str
            frequency of the data 
        '''
        self.symbol = symbol
        self.SMA_S = SMA_S
        self.SMA_L = SMA_L
        self.start = start
        self.end = end
        self.freq = freq
        self.results = None
        #add trading cost (perentace trading cost)
        self.ptc = ptc
        #once the initiatiton, the get_data function automaticly called
        self.get_data()
        self.prepare_data()
        
    #creating get data functionality
    def get_data(self):
        ''' Imports the data from oanda API (source can be changed).
        '''
        #importing oanda api
        api = tpqoa.tpqoa("oanda.cfg")
        #taking history data from oanda as we specify (and for price is 'M' or middle price (between ask and bid price)
        raw = api.get_history(instrument = self.symbol, start = self.start, end = self.end,
                granularity = self.freq, price = "M")
        #change close to price and dropna missing values
        raw = raw[['c']].copy() 
        raw = raw.dropna()
        raw.rename(columns={'c':'price'}, inplace = True)
        #create return data, sma short, sma long data base on what we specify
        raw['returns'] = np.log(raw / raw.shift(1))
        self.data = raw

    def prepare_data(self):
        '''Prepares the data for strategy backtesting (strategy-specific).
        '''
        data = self.data.copy()
        data['SMA_S'] = data['price'].rolling(self.SMA_S).mean()
        data['SMA_L'] = data['price'].rolling(self.SMA_L).mean()
        self.data = data

    #function for update a new parameters if we want 
    def set_parameters(self, SMA_S = None, SMA_L = None):
        ''' Updates SMA parameters and the prepared dataset.
        '''
        #if there's a call in this function then pass the parameter given and update the variable of SMA_S & SMA_L
        if SMA_S is not None:
            self.SMA_S = SMA_S
            self.data['SMA_S'] = self.data['price'].rolling(self.SMA_S).mean()
        if SMA_L is not None:
            self.SMA_L = SMA_L
            self.data['SMA_L'] = self.data['price'].rolling(self.SMA_L).mean()

    #testing strategy function and create the banchmark dan strategy return 
    def test_strategy(self):
        ''' Backtests the SMA-based trading strategy.
        '''
        #drop missing values
        data = self.data.copy().dropna()
        #create position and strategy return 
        #(testing happpen here) which multiply the position with return with shift(1), meaning it tell us how the prediction return which our return today is base on our prediction yesterday.
        data['position'] = np.where(data['SMA_S'] > data['SMA_L'], 1, -1)
        data['strategy'] = data['position'].shift(1) * data['returns']
        data.dropna(inplace = True)

        #trading cost variable
        data['trades'] = data.position.diff().fillna(0).abs()
        data['strategy_net'] = data.strategy - data.trades * self.ptc
        
        
        #create cumulative return of banchmark and strategy every index given
        data['creturns'] = data['returns'].cumsum().apply(np.exp)
        data['cstrategy_net'] = data.strategy_net.cumsum().apply(np.exp)
        #storing all the calculated result on results function
        self.results = data
        
        #calculatinng strategy evaluation or performance
        perf = data['cstrategy_net'].iloc[-1]#absolute performance
        #outperformance (cumulative performance of strategy - cumulative benchmark strategy) which result of how poor or good the strategy compere to the banchmark
        outperf = perf - data['creturns'].iloc[-1] 
        #return of round 6 digit of performance (1) & outperfomance of strategy compare to banchmark (2)
        return round(perf, 6), round(outperf, 6)

    #plotting the results function
    def plot_results(self):
        ''' Plots the performance of the trading strategy and compares to "buy and hold".
        '''
        if self.results is None:
            print('Run test_strategy() first.')
        else:
            #plotting the benchmark and the strategy return
            title = '{} | SMA_S = {} | SMA_L = {}'.format(self.symbol, self.SMA_S, self.SMA_L)
            self.results[['creturns', 'cstrategy_net']].plot(title = title, figsize = (12,8))

    #Strategy optimization function
    #passing the range we want for short and long SMA
    def optimize_parameters(self, SMA_S_range, SMA_L_range):
        ''' Finds the optimal strategy (global maximum) given the SMA parameter ranges.

        Parameters
        ----------
        SMA_S_range, SMA_L_range: tuple
            tuples of the form (start, end, step size)
        '''
        #unpack the range combination with (*) and store all the combination in combinations variable
        combinations = list(product( range(*SMA_S_range), range(*SMA_L_range)))

        #test all combinations
        results = []
        for comb in combinations:
            #taking short sma for [0] and long sma for [1]
            self.set_parameters(comb[0], comb[1])
            #test the strategy and add it into our list, the result of the performance strategy
            results.append(self.test_strategy()[0])
            
        #saving the best performance in variable and taking the argument or combination which resulting in best performance
        best_perf = np.max(results) #best performance
        opt = combinations[np.argmax(results)] #optimal combination parameters

        #run/set the optimal strategy
        self.set_parameters(opt[0], opt[1]) #taking both short ma and long ma in opt variable and setting the parameter accordingly
        self.test_strategy() #train the strategy with function given

        #create a df with many results
        many_results = pd.DataFrame(data = combinations, columns = ['SMA_S', 'SMA_L']) #storing all the combinations in many_results variable
        many_results['performance'] = results #storing the performance in performance column
        self.results_overview = many_results #storing the data frame into result overview function (could show it if you want to see the list_

        #returning optimal combination (opt) and the best performance result (best_perf)
        return opt, best_perf
